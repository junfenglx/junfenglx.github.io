#include <iostream>
using namespace std;
int Ca,Cb,N;
void work()
{
    int a=0,b=0;
    if(Cb==N)
        cout << "fill B" << endl;
    else
    {
        if(Ca==N)
        {
            cout << "fill A" << endl;
            cout << "pour A B" << endl;
        }
        else
        {
            while(b!=N)
            {
                if(a==0)
                {
                    cout << "fill A" << endl;
                    a=Ca;
                }
                else if(b==Cb)
                {
                    b=a;
                    cout << "empty B" << endl;
                    a=0;
                    cout << "pour A B" << endl;
                }
                else if(a+b>Cb)
                {
                    a-=Cb-b;
                    b=Cb;
                    cout << "pour A B" << endl;
                }
                else
                {
                    b+=a;
                    a=0;
                    cout << "pour A B" << endl;
                }
            }
        }
    }
    cout << "success" << endl;
}
int main()
{
    while(cin >> Ca >> Cb >> N)
    {
        work();
    }
    return 0;
}
